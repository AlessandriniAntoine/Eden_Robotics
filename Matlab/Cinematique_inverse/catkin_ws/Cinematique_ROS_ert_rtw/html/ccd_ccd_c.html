105void __ccdSupport(const void *obj1, const void *obj2,



106                  const ccd_vec3_t *_dir, const ccd_t *ccd,



107                  ccd_support_t *supp)



108{



109    ccd_vec3_t dir;



110



111    ccdVec3Copy(&dir, _dir);



112



113    ccd->support1(obj1, &dir, &supp->v1);



114



115    ccdVec3Scale(&dir, -CCD_ONE);



116    ccd->support2(obj2, &dir, &supp->v2);



117



118    ccdVec3Sub2(&supp->v, &supp->v1, &supp->v2);



119}

121int ccdGJKIntersect(const void *obj1, const void *obj2, const ccd_t *ccd)



122{



123    ccd_simplex_t simplex;



124    return __ccdGJK(obj1, obj2, ccd, &simplex) == 0;



125}

127int ccdGJKSeparate(const void *obj1, const void *obj2, const ccd_t *ccd,



128                   ccd_vec3_t *sep)



129{



130    ccd_pt_t polytope;



131    ccd_pt_el_t *nearest;



132    int ret;



133



134    ccdPtInit(&polytope);



135



136    ret = __ccdGJKEPA(obj1, obj2, ccd, &polytope, &nearest);



137



138    // set separation vector



139    if (nearest)



140        ccdVec3Copy(sep, &nearest->witness);



141



142    ccdPtDestroy(&polytope);



143



144    return ret;



145}

148static int penEPAPosCmp(const void *a, const void *b)



149{



150    ccd_pt_vertex_t *v1, *v2;



151    v1 = *(ccd_pt_vertex_t **)a;



152    v2 = *(ccd_pt_vertex_t **)b;



153



154    if (ccdEq(v1->dist, v2->dist)){



155        return 0;



156    }else if (v1->dist < v2->dist){



157        return -1;



158    }else{



159        return 1;



160    }



161}



162



163static int penEPAPos(const ccd_pt_t *pt, const ccd_pt_el_t *nearest,



164                     ccd_vec3_t *pos)



165{



166    (void)nearest;//C4100 in MSVC and -Wunused-value in Clang



167    ccd_pt_vertex_t *v;



168    ccd_pt_vertex_t **vs;



169    size_t i, len;



170    ccd_real_t scale;



171



172    // compute median



173    len = 0;



174    ccdListForEachEntry(&pt->vertices, v, ccd_pt_vertex_t, list){



175        len++;



176    }



177



178    vs = CCD_ALLOC_ARR(ccd_pt_vertex_t *, len);



179    if (vs == NULL)



180        return -1;



181



182    i = 0;



183    ccdListForEachEntry(&pt->vertices, v, ccd_pt_vertex_t, list){



184        vs[i++] = v;



185    }



186



187    qsort(vs, len, sizeof(ccd_pt_vertex_t *), penEPAPosCmp);



188



189    ccdVec3Set(pos, CCD_ZERO, CCD_ZERO, CCD_ZERO);



190    scale = CCD_ZERO;



191    if (len % 2 == 1)



192        len++;



193



194    for (i = 0; i < len / 2; i++){



195        ccdVec3Add(pos, &vs[i]->v.v1);



196        ccdVec3Add(pos, &vs[i]->v.v2);



197        scale += CCD_REAL(2.);



198    }



199    ccdVec3Scale(pos, CCD_ONE / scale);



200



201    free(vs);



202



203    return 0;



204}

206int ccdGJKPenetration(const void *obj1, const void *obj2, const ccd_t *ccd,



207                      ccd_real_t *depth, ccd_vec3_t *dir, ccd_vec3_t *pos)



208{



209    ccd_pt_t polytope;



210    ccd_pt_el_t *nearest;



211    int ret;



212



213    ccdPtInit(&polytope);



214



215    ret = __ccdGJKEPA(obj1, obj2, ccd, &polytope, &nearest);



216



217    // set separation vector



218    if (ret == 0 && nearest){



219        // compute depth of penetration



220        *depth = CCD_SQRT(nearest->dist);



221



222        // store normalized direction vector



223        ccdVec3Copy(dir, &nearest->witness);



224        ccdVec3Normalize(dir);



225



226        // compute position



227        if (penEPAPos(&polytope, nearest, pos) != 0){



228            ccdPtDestroy(&polytope);



229            return -2;



230        }



231    }



232



233    ccdPtDestroy(&polytope);



234



235    return ret;



236}

239int __ccdGJK(const void *obj1, const void *obj2,



240                    const ccd_t *ccd, ccd_simplex_t *simplex)



241{



242    unsigned long iterations;



243    ccd_vec3_t dir; // direction vector



244    ccd_support_t last; // last support point



245    int do_simplex_res;



246



247    // initialize simplex struct



248    ccdSimplexInit(simplex);



249



250    // get first direction



251    ccd->first_dir(obj1, obj2, &dir);



252    // get first support point



253    __ccdSupport(obj1, obj2, &dir, ccd, &last);



254    // and add this point to simplex as last one



255    ccdSimplexAdd(simplex, &last);



256



257    // set up direction vector to as (O - last) which is exactly -last



258    ccdVec3Copy(&dir, &last.v);



259    ccdVec3Scale(&dir, -CCD_ONE);



260



261    // start iterations



262    for (iterations = 0UL; iterations < ccd->max_iterations; ++iterations) {



263        // obtain support point



264        __ccdSupport(obj1, obj2, &dir, ccd, &last);



265



266        // check if farthest point in Minkowski difference in direction dir



267        // isn't somewhere before origin (the test on negative dot product)



268        // - because if it is, objects are not intersecting at all.



269        if (ccdVec3Dot(&last.v, &dir) < CCD_ZERO){



270            return -1; // intersection not found



271        }



272



273        // add last support vector to simplex



274        ccdSimplexAdd(simplex, &last);



275



276        // if doSimplex returns 1 if objects intersect, -1 if objects don't



277        // intersect and 0 if algorithm should continue



278        do_simplex_res = doSimplex(simplex, &dir);



279        if (do_simplex_res == 1){



280            return 0; // intersection found



281        }else if (do_simplex_res == -1){



282            return -1; // intersection not found



283        }



284



285        if (ccdIsZero(ccdVec3Len2(&dir))){



286            return -1; // intersection not found



287        }



288    }



289



290    // intersection wasn't found



291    return -1;



292}

294static int __ccdGJKEPA(const void *obj1, const void *obj2,



295                       const ccd_t *ccd,



296                       ccd_pt_t *polytope, ccd_pt_el_t **nearest)



297{



298    ccd_simplex_t simplex;



299    ccd_support_t supp; // support point



300    int ret, size;



301



302    *nearest = NULL;



303



304    // run GJK and obtain terminal simplex



305    ret = __ccdGJK(obj1, obj2, ccd, &simplex);



306    if (ret != 0)



307        return -1;



308



309    // transform simplex to polytope - simplex won't be used anymore



310    size = ccdSimplexSize(&simplex);



311    if (size == 4){



312        ret = simplexToPolytope4(obj1, obj2, ccd, &simplex, polytope, nearest);



313    }else if (size == 3){



314        ret = simplexToPolytope3(obj1, obj2, ccd, &simplex, polytope, nearest);



315    }else{ // size == 2



316        ret = simplexToPolytope2(obj1, obj2, ccd, &simplex, polytope, nearest);



317    }



318



319    if (ret == -1){



320        // touching contact



321        return 0;



322    }else if (ret == -2){



323        // failed memory allocation



324        return -2;



325    }



326



327



328    while (1){



329        // get triangle nearest to origin



330        *nearest = ccdPtNearest(polytope);



331



332        // get next support point



333        if (nextSupport(obj1, obj2, ccd, *nearest, &supp) != 0)



334            break;



335



336        // expand nearest triangle using new point - supp



337        if (expandPolytope(polytope, *nearest, &supp) != 0)



338            return -2;



339    }



340



341    return 0;



342}

346static int doSimplex2(ccd_simplex_t *simplex, ccd_vec3_t *dir)



347{



348    const ccd_support_t *A, *B;



349    ccd_vec3_t AB, AO, tmp;



350    ccd_real_t dot;



351



352    // get last added as A



353    A = ccdSimplexLast(simplex);



354    // get the other point



355    B = ccdSimplexPoint(simplex, 0);



356    // compute AB oriented segment



357    ccdVec3Sub2(&AB, &B->v, &A->v);



358    // compute AO vector



359    ccdVec3Copy(&AO, &A->v);



360    ccdVec3Scale(&AO, -CCD_ONE);



361



362    // dot product AB . AO



363    dot = ccdVec3Dot(&AB, &AO);



364



365    // check if origin doesn't lie on AB segment



366    ccdVec3Cross(&tmp, &AB, &AO);



367    if (ccdIsZero(ccdVec3Len2(&tmp)) && dot > CCD_ZERO){



368        return 1;



369    }



370



371    // check if origin is in area where AB segment is



372    if (ccdIsZero(dot) || dot < CCD_ZERO){



373        // origin is in outside are of A



374        ccdSimplexSet(simplex, 0, A);



375        ccdSimplexSetSize(simplex, 1);



376        ccdVec3Copy(dir, &AO);



377    }else{



378        // origin is in area where AB segment is



379



380        // keep simplex untouched and set direction to



381        // AB x AO x AB



382        tripleCross(&AB, &AO, &AB, dir);



383    }



384



385    return 0;



386}

388static int doSimplex3(ccd_simplex_t *simplex, ccd_vec3_t *dir)



389{



390    const ccd_support_t *A, *B, *C;



391    ccd_vec3_t AO, AB, AC, ABC, tmp;



392    ccd_real_t dot, dist;



393



394    // get last added as A



395    A = ccdSimplexLast(simplex);



396    // get the other points



397    B = ccdSimplexPoint(simplex, 1);



398    C = ccdSimplexPoint(simplex, 0);



399



400    // check touching contact



401    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &A->v, &B->v, &C->v, NULL);



402



403    // MathWorks specific comment:



404    // So here is the issue: when libccd compiled with single precision, ccdIsZero uses FLT_EPSILON as the tolerance threshold (around 1.19E-7), and



405    // the variable 'dist' here is actually dist squared. So if the absolute value of the actual distance is smaller than 0.001, this 'dist' variable



406    // will fall below the tolerance threshold, as a result, the simplex will be recognized as containing the origin and the two objects will be



407    // considered touching each other. I feel here ccdIsZero(CCD_SQRT(dist)) might be more appropriate, but the author might have more considerations



408    // on numerical stability, and this pattern has been used throughout the libccd source code, so I tend to not change it.



409    // Compiling libccd with double precission will bring the minimal detectable separation distance down to below 1E-6, at the cost of slower speed



410    if (ccdIsZero(dist)){



411        return 1;



412    }



413



414    // check if triangle is really triangle (has area > 0)



415    // if not simplex can't be expanded and thus no itersection is found



416    if (ccdVec3Eq(&A->v, &B->v) || ccdVec3Eq(&A->v, &C->v)){



417        return -1;



418    }



419



420    // compute AO vector



421    ccdVec3Copy(&AO, &A->v);



422    ccdVec3Scale(&AO, -CCD_ONE);



423



424    // compute AB and AC segments and ABC vector (perpendircular to triangle)



425    ccdVec3Sub2(&AB, &B->v, &A->v);



426    ccdVec3Sub2(&AC, &C->v, &A->v);



427    ccdVec3Cross(&ABC, &AB, &AC);



428



429    ccdVec3Cross(&tmp, &ABC, &AC);



430    dot = ccdVec3Dot(&tmp, &AO);



431    if (ccdIsZero(dot) || dot > CCD_ZERO){



432        dot = ccdVec3Dot(&AC, &AO);



433        if (ccdIsZero(dot) || dot > CCD_ZERO){



434            // C is already in place



435            ccdSimplexSet(simplex, 1, A);



436            ccdSimplexSetSize(simplex, 2);



437            tripleCross(&AC, &AO, &AC, dir);



438        }else{



439ccd_do_simplex3_45:



440            dot = ccdVec3Dot(&AB, &AO);



441            if (ccdIsZero(dot) || dot > CCD_ZERO){



442                ccdSimplexSet(simplex, 0, B);



443                ccdSimplexSet(simplex, 1, A);



444                ccdSimplexSetSize(simplex, 2);



445                tripleCross(&AB, &AO, &AB, dir);



446            }else{



447                ccdSimplexSet(simplex, 0, A);



448                ccdSimplexSetSize(simplex, 1);



449                ccdVec3Copy(dir, &AO);



450            }



451        }



452    }else{



453        ccdVec3Cross(&tmp, &AB, &ABC);



454        dot = ccdVec3Dot(&tmp, &AO);



455        if (ccdIsZero(dot) || dot > CCD_ZERO){



456            goto ccd_do_simplex3_45;



457        }else{



458            dot = ccdVec3Dot(&ABC, &AO);



459            if (ccdIsZero(dot) || dot > CCD_ZERO){



460                ccdVec3Copy(dir, &ABC);



461            }else{



462                ccd_support_t Ctmp;



463                ccdSupportCopy(&Ctmp, C);



464                ccdSimplexSet(simplex, 0, B);



465                ccdSimplexSet(simplex, 1, &Ctmp);



466



467                ccdVec3Copy(dir, &ABC);



468                ccdVec3Scale(dir, -CCD_ONE);



469            }



470        }



471    }



472



473    return 0;



474}

476static int doSimplex4(ccd_simplex_t *simplex, ccd_vec3_t *dir)



477{



478    const ccd_support_t *A, *B, *C, *D;



479    ccd_vec3_t AO, AB, AC, AD, ABC, ACD, ADB;



480    int B_on_ACD, C_on_ADB, D_on_ABC;



481    int AB_O, AC_O, AD_O;



482    ccd_real_t dist;



483



484    // get last added as A



485    A = ccdSimplexLast(simplex);



486    // get the other points



487    B = ccdSimplexPoint(simplex, 2);



488    C = ccdSimplexPoint(simplex, 1);



489    D = ccdSimplexPoint(simplex, 0);



490



491    // check if tetrahedron is really tetrahedron (has volume > 0)



492    // if it is not simplex can't be expanded and thus no intersection is



493    // found



494    dist = ccdVec3PointTriDist2(&A->v, &B->v, &C->v, &D->v, NULL);



495    if (ccdIsZero(dist)){



496        return -1;



497    }



498



499    // check if origin lies on some of tetrahedron's face - if so objects



500    // intersect



501    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &A->v, &B->v, &C->v, NULL);



502    if (ccdIsZero(dist))



503        return 1;



504    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &A->v, &C->v, &D->v, NULL);



505    if (ccdIsZero(dist))



506        return 1;



507    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &A->v, &B->v, &D->v, NULL);



508    if (ccdIsZero(dist))



509        return 1;



510    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &B->v, &C->v, &D->v, NULL);



511    if (ccdIsZero(dist))



512        return 1;



513



514    // compute AO, AB, AC, AD segments and ABC, ACD, ADB normal vectors



515    ccdVec3Copy(&AO, &A->v);



516    ccdVec3Scale(&AO, -CCD_ONE);



517    ccdVec3Sub2(&AB, &B->v, &A->v);



518    ccdVec3Sub2(&AC, &C->v, &A->v);



519    ccdVec3Sub2(&AD, &D->v, &A->v);



520    ccdVec3Cross(&ABC, &AB, &AC);



521    ccdVec3Cross(&ACD, &AC, &AD);



522    ccdVec3Cross(&ADB, &AD, &AB);



523



524    // side (positive or negative) of B, C, D relative to planes ACD, ADB



525    // and ABC respectively



526    B_on_ACD = ccdSign(ccdVec3Dot(&ACD, &AB));



527    C_on_ADB = ccdSign(ccdVec3Dot(&ADB, &AC));



528    D_on_ABC = ccdSign(ccdVec3Dot(&ABC, &AD));



529



530    // whether origin is on same side of ACD, ADB, ABC as B, C, D



531    // respectively



532    AB_O = ccdSign(ccdVec3Dot(&ACD, &AO)) == B_on_ACD;



533    AC_O = ccdSign(ccdVec3Dot(&ADB, &AO)) == C_on_ADB;



534    AD_O = ccdSign(ccdVec3Dot(&ABC, &AO)) == D_on_ABC;



535



536    if (AB_O && AC_O && AD_O){



537        // origin is in tetrahedron



538        return 1;



539



540    // rearrange simplex to triangle and call doSimplex3()



541    }else if (!AB_O){



542        // B is farthest from the origin among all of the tetrahedron's



543        // points, so remove it from the list and go on with the triangle



544        // case



545



546        // D and C are in place



547        ccdSimplexSet(simplex, 2, A);



548        ccdSimplexSetSize(simplex, 3);



549    }else if (!AC_O){



550        // C is farthest



551        ccdSimplexSet(simplex, 1, D);



552        ccdSimplexSet(simplex, 0, B);



553        ccdSimplexSet(simplex, 2, A);



554        ccdSimplexSetSize(simplex, 3);



555    }else{ // (!AD_O)



556        ccdSimplexSet(simplex, 0, C);



557        ccdSimplexSet(simplex, 1, B);



558        ccdSimplexSet(simplex, 2, A);



559        ccdSimplexSetSize(simplex, 3);



560    }



561



562    return doSimplex3(simplex, dir);



563}



564



565static int doSimplex(ccd_simplex_t *simplex, ccd_vec3_t *dir)



566{



567    if (ccdSimplexSize(simplex) == 2){



568        // simplex contains segment only one segment



569        return doSimplex2(simplex, dir);



570    }else if (ccdSimplexSize(simplex) == 3){



571        // simplex contains triangle



572        return doSimplex3(simplex, dir);



573    }else{ // ccdSimplexSize(simplex) == 4



574        // tetrahedron - this is the only shape which can encapsule origin



575        // so doSimplex4() also contains test on it



576        return doSimplex4(simplex, dir);



577    }



578}

581_ccd_inline void tripleCross(const ccd_vec3_t *a, const ccd_vec3_t *b,



582                             const ccd_vec3_t *c, ccd_vec3_t *d)

586    ccdVec3Cross(d, &e, c);

591/** Transforms simplex to polytope. It is assumed that simplex has 4



592 *  vertices! */



593static int simplexToPolytope4(const void *obj1, const void *obj2,



594                              const ccd_t *ccd,



595                              ccd_simplex_t *simplex,



596                              ccd_pt_t *pt, ccd_pt_el_t **nearest)



597{



598    const ccd_support_t *a, *b, *c, *d;



599    int use_polytope3;



600    ccd_real_t dist;



601    ccd_pt_vertex_t *v[4];



602    ccd_pt_edge_t *e[6];



603    int i;



604



605    a = ccdSimplexPoint(simplex, 0);



606    b = ccdSimplexPoint(simplex, 1);



607    c = ccdSimplexPoint(simplex, 2);



608    d = ccdSimplexPoint(simplex, 3);



609



610    // check if origin lies on some of tetrahedron's face - if so use



611    // simplexToPolytope3()



612    use_polytope3 = 0;



613    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &a->v, &b->v, &c->v, NULL);



614    if (ccdIsZero(dist)){



615        use_polytope3 = 1;



616    }



617    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &a->v, &c->v, &d->v, NULL);



618    if (ccdIsZero(dist)){



619        use_polytope3 = 1;



620        ccdSimplexSet(simplex, 1, c);



621        ccdSimplexSet(simplex, 2, d);



622    }



623    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &a->v, &b->v, &d->v, NULL);



624    if (ccdIsZero(dist)){



625        use_polytope3 = 1;



626        ccdSimplexSet(simplex, 2, d);



627    }



628    dist = ccdVec3PointTriDist2(ccd_vec3_origin, &b->v, &c->v, &d->v, NULL);



629    if (ccdIsZero(dist)){



630        use_polytope3 = 1;



631        ccdSimplexSet(simplex, 0, b);



632        ccdSimplexSet(simplex, 1, c);



633        ccdSimplexSet(simplex, 2, d);



634    }



635



636    if (use_polytope3){



637        ccdSimplexSetSize(simplex, 3);



638        return simplexToPolytope3(obj1, obj2, ccd, simplex, pt, nearest);



639    }



640



641    // no touching contact - simply create tetrahedron



642    for (i = 0; i < 4; i++){



643        v[i] = ccdPtAddVertex(pt, ccdSimplexPoint(simplex, i));



644    }



645    



646    e[0] = ccdPtAddEdge(pt, v[0], v[1]);



647    e[1] = ccdPtAddEdge(pt, v[1], v[2]);



648    e[2] = ccdPtAddEdge(pt, v[2], v[0]);



649    e[3] = ccdPtAddEdge(pt, v[3], v[0]);



650    e[4] = ccdPtAddEdge(pt, v[3], v[1]);



651    e[5] = ccdPtAddEdge(pt, v[3], v[2]);



652



653    // ccdPtAdd*() functions return NULL either if the memory allocation



654    // failed of if any of the input pointers are NULL, so the bad



655    // allocation can be checked by the last calls of ccdPtAddFace()



656    // because the rest of the bad allocations eventually "bubble up" here



657    if (ccdPtAddFace(pt, e[0], e[1], e[2]) == NULL



658            || ccdPtAddFace(pt, e[3], e[4], e[0]) == NULL



659            || ccdPtAddFace(pt, e[4], e[5], e[1]) == NULL



660            || ccdPtAddFace(pt, e[5], e[3], e[2]) == NULL){



661        return -2;



662    }



663



664    return 0;



665}

667/** Transforms simplex to polytope, three vertices required */



668static int simplexToPolytope3(const void *obj1, const void *obj2,



669                              const ccd_t *ccd,



670                              const ccd_simplex_t *simplex,



671                              ccd_pt_t *pt, ccd_pt_el_t **nearest)



672{



673    const ccd_support_t *a, *b, *c;



674    ccd_support_t d, d2;



675    ccd_vec3_t ab, ac, dir;



676    ccd_pt_vertex_t *v[5];



677    ccd_pt_edge_t *e[9];



678    ccd_real_t dist, dist2;



679



680    *nearest = NULL;



681



682    a = ccdSimplexPoint(simplex, 0);



683    b = ccdSimplexPoint(simplex, 1);



684    c = ccdSimplexPoint(simplex, 2);



685



686    // If only one triangle left from previous GJK run origin lies on this



687    // triangle. So it is necessary to expand triangle into two



688    // tetrahedrons connected with base (which is exactly abc triangle).



689



690    // get next support point in direction of normal of triangle



691    ccdVec3Sub2(&ab, &b->v, &a->v);



692    ccdVec3Sub2(&ac, &c->v, &a->v);



693    ccdVec3Cross(&dir, &ab, &ac);



694    __ccdSupport(obj1, obj2, &dir, ccd, &d);



695    dist = ccdVec3PointTriDist2(&d.v, &a->v, &b->v, &c->v, NULL);



696



697    // and second one take in opposite direction



698    ccdVec3Scale(&dir, -CCD_ONE);



699    __ccdSupport(obj1, obj2, &dir, ccd, &d2);



700    dist2 = ccdVec3PointTriDist2(&d2.v, &a->v, &b->v, &c->v, NULL);



701



702    // check if face isn't already on edge of minkowski sum and thus we



703    // have touching contact



704    if (ccdIsZero(dist) || ccdIsZero(dist2)){



705        v[0] = ccdPtAddVertex(pt, a);



706        v[1] = ccdPtAddVertex(pt, b);



707        v[2] = ccdPtAddVertex(pt, c);



708        e[0] = ccdPtAddEdge(pt, v[0], v[1]);



709        e[1] = ccdPtAddEdge(pt, v[1], v[2]);



710        e[2] = ccdPtAddEdge(pt, v[2], v[0]);



711        *nearest = (ccd_pt_el_t *)ccdPtAddFace(pt, e[0], e[1], e[2]);



712        if (*nearest == NULL)



713            return -2;



714



715        return -1;



716    }



717



718    // form polyhedron



719    v[0] = ccdPtAddVertex(pt, a);



720    v[1] = ccdPtAddVertex(pt, b);



721    v[2] = ccdPtAddVertex(pt, c);



722    v[3] = ccdPtAddVertex(pt, &d);



723    v[4] = ccdPtAddVertex(pt, &d2);



724



725    e[0] = ccdPtAddEdge(pt, v[0], v[1]);



726    e[1] = ccdPtAddEdge(pt, v[1], v[2]);



727    e[2] = ccdPtAddEdge(pt, v[2], v[0]);



728



729    e[3] = ccdPtAddEdge(pt, v[3], v[0]);



730    e[4] = ccdPtAddEdge(pt, v[3], v[1]);



731    e[5] = ccdPtAddEdge(pt, v[3], v[2]);



732



733    e[6] = ccdPtAddEdge(pt, v[4], v[0]);



734    e[7] = ccdPtAddEdge(pt, v[4], v[1]);



735    e[8] = ccdPtAddEdge(pt, v[4], v[2]);



736



737    if (ccdPtAddFace(pt, e[3], e[4], e[0]) == NULL



738            || ccdPtAddFace(pt, e[4], e[5], e[1]) == NULL



739            || ccdPtAddFace(pt, e[5], e[3], e[2]) == NULL



740



741            || ccdPtAddFace(pt, e[6], e[7], e[0]) == NULL



742            || ccdPtAddFace(pt, e[7], e[8], e[1]) == NULL



743            || ccdPtAddFace(pt, e[8], e[6], e[2]) == NULL){



744        return -2;



745    }



746



747    return 0;



748}

750/** Transforms simplex to polytope, two vertices required */



751static int simplexToPolytope2(const void *obj1, const void *obj2,



752                              const ccd_t *ccd,



753                              const ccd_simplex_t *simplex,



754                              ccd_pt_t *pt, ccd_pt_el_t **nearest)



755{



756    const ccd_support_t *a, *b;



757    ccd_vec3_t ab, ac, dir;



758    ccd_support_t supp[4];



759    ccd_pt_vertex_t *v[6];



760    ccd_pt_edge_t *e[12];



761    size_t i;



762    int found;



763



764    a = ccdSimplexPoint(simplex, 0);



765    b = ccdSimplexPoint(simplex, 1);



766



767    // This situation is a bit tricky. If only one segment comes from



768    // previous run of GJK - it means that either this segment is on



769    // minkowski edge (and thus we have touch contact) or it it isn't and



770    // therefore segment is somewhere *inside* minkowski sum and it *must*



771    // be possible to fully enclose this segment with polyhedron formed by



772    // at least 8 triangle faces.



773



774    // get first support point (any)



775    found = 0;



776    for (i = 0; i < ccd_points_on_sphere_len; i++){



777        __ccdSupport(obj1, obj2, &ccd_points_on_sphere[i], ccd, &supp[0]);



778        if (!ccdVec3Eq(&a->v, &supp[0].v) && !ccdVec3Eq(&b->v, &supp[0].v)){



779            found = 1;



780            break;



781        }



782    }



783    if (!found)



784        goto simplexToPolytope2_touching_contact;



785



786    // get second support point in opposite direction than supp[0]



787    ccdVec3Copy(&dir, &supp[0].v);



788    ccdVec3Scale(&dir, -CCD_ONE);



789    __ccdSupport(obj1, obj2, &dir, ccd, &supp[1]);



790    if (ccdVec3Eq(&a->v, &supp[1].v) || ccdVec3Eq(&b->v, &supp[1].v))



791        goto simplexToPolytope2_touching_contact;



792



793    // next will be in direction of normal of triangle a,supp[0],supp[1]



794    ccdVec3Sub2(&ab, &supp[0].v, &a->v);



795    ccdVec3Sub2(&ac, &supp[1].v, &a->v);



796    ccdVec3Cross(&dir, &ab, &ac);



797    __ccdSupport(obj1, obj2, &dir, ccd, &supp[2]);



798    if (ccdVec3Eq(&a->v, &supp[2].v) || ccdVec3Eq(&b->v, &supp[2].v))



799        goto simplexToPolytope2_touching_contact;



800



801    // and last one will be in opposite direction



802    ccdVec3Scale(&dir, -CCD_ONE);



803    __ccdSupport(obj1, obj2, &dir, ccd, &supp[3]);



804    if (ccdVec3Eq(&a->v, &supp[3].v) || ccdVec3Eq(&b->v, &supp[3].v))



805        goto simplexToPolytope2_touching_contact;



806



807    goto simplexToPolytope2_not_touching_contact;



808simplexToPolytope2_touching_contact:



809    v[0] = ccdPtAddVertex(pt, a);



810    v[1] = ccdPtAddVertex(pt, b);



811    *nearest = (ccd_pt_el_t *)ccdPtAddEdge(pt, v[0], v[1]);



812    if (*nearest == NULL)



813        return -2;



814



815    return -1;



816



817simplexToPolytope2_not_touching_contact:



818    // form polyhedron



819    v[0] = ccdPtAddVertex(pt, a);



820    v[1] = ccdPtAddVertex(pt, &supp[0]);



821    v[2] = ccdPtAddVertex(pt, b);



822    v[3] = ccdPtAddVertex(pt, &supp[1]);



823    v[4] = ccdPtAddVertex(pt, &supp[2]);



824    v[5] = ccdPtAddVertex(pt, &supp[3]);



825



826    e[0] = ccdPtAddEdge(pt, v[0], v[1]);



827    e[1] = ccdPtAddEdge(pt, v[1], v[2]);



828    e[2] = ccdPtAddEdge(pt, v[2], v[3]);



829    e[3] = ccdPtAddEdge(pt, v[3], v[0]);



830



831    e[4] = ccdPtAddEdge(pt, v[4], v[0]);



832    e[5] = ccdPtAddEdge(pt, v[4], v[1]);



833    e[6] = ccdPtAddEdge(pt, v[4], v[2]);



834    e[7] = ccdPtAddEdge(pt, v[4], v[3]);



835



836    e[8]  = ccdPtAddEdge(pt, v[5], v[0]);



837    e[9]  = ccdPtAddEdge(pt, v[5], v[1]);



838    e[10] = ccdPtAddEdge(pt, v[5], v[2]);



839    e[11] = ccdPtAddEdge(pt, v[5], v[3]);



840



841    if (ccdPtAddFace(pt, e[4], e[5], e[0]) == NULL



842            || ccdPtAddFace(pt, e[5], e[6], e[1]) == NULL



843            || ccdPtAddFace(pt, e[6], e[7], e[2]) == NULL



844            || ccdPtAddFace(pt, e[7], e[4], e[3]) == NULL



845



846            || ccdPtAddFace(pt, e[8],  e[9],  e[0]) == NULL



847            || ccdPtAddFace(pt, e[9],  e[10], e[1]) == NULL



848            || ccdPtAddFace(pt, e[10], e[11], e[2]) == NULL



849            || ccdPtAddFace(pt, e[11], e[8],  e[3]) == NULL){



850        return -2;



851    }



852



853    return 0;



854}

856/** Expands polytope's tri by new vertex v. Triangle tri is replaced by



857 *  three triangles each with one vertex in v. */



858static int expandPolytope(ccd_pt_t *pt, ccd_pt_el_t *el,



859                          const ccd_support_t *newv)



860{



861    ccd_pt_vertex_t *v[5];



862    ccd_pt_edge_t *e[8];



863    ccd_pt_face_t *f[2];



864



865



866    // element can be either segment or triangle



867    if (el->type == CCD_PT_EDGE){



868        // In this case, segment should be replaced by new point.



869        // Simpliest case is when segment stands alone and in this case



870        // this segment is replaced by two other segments both connected to



871        // newv.



872        // Segment can be also connected to max two faces and in that case



873        // each face must be replaced by two other faces. To do this



874        // correctly it is necessary to have correctly ordered edges and



875        // vertices which is exactly what is done in following code.



876        //



877



878        ccdPtEdgeVertices((const ccd_pt_edge_t *)el, &v[0], &v[2]);



879



880        ccdPtEdgeFaces((ccd_pt_edge_t *)el, &f[0], &f[1]);



881



882        if (f[0]){



883            ccdPtFaceEdges(f[0], &e[0], &e[1], &e[2]);



884            if (e[0] == (ccd_pt_edge_t *)el){



885                e[0] = e[2];



886            }else if (e[1] == (ccd_pt_edge_t *)el){



887                e[1] = e[2];



888            }



889            ccdPtEdgeVertices(e[0], &v[1], &v[3]);



890            if (v[1] != v[0] && v[3] != v[0]){



891                e[2] = e[0];



892                e[0] = e[1];



893                e[1] = e[2];



894                if (v[1] == v[2])



895                    v[1] = v[3];



896            }else{



897                if (v[1] == v[0])



898                    v[1] = v[3];



899            }



900



901            if (f[1]){



902                ccdPtFaceEdges(f[1], &e[2], &e[3], &e[4]);



903                if (e[2] == (ccd_pt_edge_t *)el){



904                    e[2] = e[4];



905                }else if (e[3] == (ccd_pt_edge_t *)el){



906                    e[3] = e[4];



907                }



908                ccdPtEdgeVertices(e[2], &v[3], &v[4]);



909                if (v[3] != v[2] && v[4] != v[2]){



910                    e[4] = e[2];



911                    e[2] = e[3];



912                    e[3] = e[4];



913                    if (v[3] == v[0])



914                        v[3] = v[4];



915                }else{



916                    if (v[3] == v[2])



917                        v[3] = v[4];



918                }



919            }



920



921



922            v[4] = ccdPtAddVertex(pt, newv);



923



924            ccdPtDelFace(pt, f[0]);



925            if (f[1]){



926                ccdPtDelFace(pt, f[1]);



927                ccdPtDelEdge(pt, (ccd_pt_edge_t *)el);



928            }



929



930            e[4] = ccdPtAddEdge(pt, v[4], v[2]);



931            e[5] = ccdPtAddEdge(pt, v[4], v[0]);



932            e[6] = ccdPtAddEdge(pt, v[4], v[1]);



933            if (f[1])



934                e[7] = ccdPtAddEdge(pt, v[4], v[3]);



935



936



937            if (ccdPtAddFace(pt, e[1], e[4], e[6]) == NULL



938                    || ccdPtAddFace(pt, e[0], e[6], e[5]) == NULL){



939                return -2;



940            }



941



942            if (f[1]){



943                if (ccdPtAddFace(pt, e[3], e[5], e[7]) == NULL



944                        || ccdPtAddFace(pt, e[4], e[7], e[2]) == NULL){



945                    return -2;



946                }



947            }else{



948                if (ccdPtAddFace(pt, e[4], e[5], (ccd_pt_edge_t *)el) == NULL)



949                    return -2;



950            }



951        }



952    }else{ // el->type == CCD_PT_FACE



953        // replace triangle by tetrahedron without base (base would be the



954        // triangle that will be removed)



955



956        // get triplet of surrounding edges and vertices of triangle face



957        ccdPtFaceEdges((const ccd_pt_face_t *)el, &e[0], &e[1], &e[2]);



958        ccdPtEdgeVertices(e[0], &v[0], &v[1]);



959        ccdPtEdgeVertices(e[1], &v[2], &v[3]);



960



961        // following code sorts edges to have e[0] between vertices 0-1,



962        // e[1] between 1-2 and e[2] between 2-0



963        if (v[2] != v[1] && v[3] != v[1]){



964            // swap e[1] and e[2] 



965            e[3] = e[1];



966            e[1] = e[2];



967            e[2] = e[3];



968        }



969        if (v[3] != v[0] && v[3] != v[1])



970            v[2] = v[3];



971



972        // remove triangle face



973        ccdPtDelFace(pt, (ccd_pt_face_t *)el);



974



975        // expand triangle to tetrahedron



976        v[3] = ccdPtAddVertex(pt, newv);



977        e[3] = ccdPtAddEdge(pt, v[3], v[0]);



978        e[4] = ccdPtAddEdge(pt, v[3], v[1]);



979        e[5] = ccdPtAddEdge(pt, v[3], v[2]);



980



981        if (ccdPtAddFace(pt, e[3], e[4], e[0]) == NULL



982                || ccdPtAddFace(pt, e[4], e[5], e[1]) == NULL



983                || ccdPtAddFace(pt, e[5], e[3], e[2]) == NULL){



984            return -2;



985        }



986    }



987



988    return 0;



989}

991/** Finds next support point (and stores it in out argument).



992 *  Returns 0 on success, -1 otherwise */



993static int nextSupport(const void *obj1, const void *obj2, const ccd_t *ccd,



994                       const ccd_pt_el_t *el,



995                       ccd_support_t *out)



996{



997    ccd_vec3_t *a, *b, *c;



998    ccd_real_t dist;



999



1000    if (el->type == CCD_PT_VERTEX)



1001        return -1;



1002



1003    // touch contact



1004    if (ccdIsZero(el->dist))



1005        return -1;



1006



1007    __ccdSupport(obj1, obj2, &el->witness, ccd, out);



1008



1009    // Compute dist of support point along element witness point direction



1010    // so we can determine whether we expanded a polytope surrounding the



1011    // origin a bit.



1012    dist = ccdVec3Dot(&out->v, &el->witness);



1013



1014    if (dist - el->dist < ccd->epa_tolerance)



1015        return -1;



1016



1017    if (el->type == CCD_PT_EDGE){



1018        // fetch end points of edge



1019        ccdPtEdgeVec3((ccd_pt_edge_t *)el, &a, &b);



1020



1021        // get distance from segment



1022        dist = ccdVec3PointSegmentDist2(&out->v, a, b, NULL);



1023    }else{ // el->type == CCD_PT_FACE



1024        // fetch vertices of triangle face



1025        ccdPtFaceVec3((ccd_pt_face_t *)el, &a, &b, &c);



1026



1027        // check if new point can significantly expand polytope



1028        dist = ccdVec3PointTriDist2(&out->v, a, b, c, NULL);



1029    }



1030



1031    if (dist < ccd->epa_tolerance)



1032        return -1;



1033



1034    return 0;



1035}
