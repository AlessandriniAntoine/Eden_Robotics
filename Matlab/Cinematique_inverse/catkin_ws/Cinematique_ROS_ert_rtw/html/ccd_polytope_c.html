35_ccd_inline void _ccdPtNearestUpdate(ccd_pt_t *pt, ccd_pt_el_t *el)



36{



37    if (ccdEq(pt->nearest_dist, el->dist)){



38        if (el->type < pt->nearest_type){



39            pt->nearest = el;



40            pt->nearest_dist = el->dist;



41            pt->nearest_type = el->type;



42        }



43    }else if (el->dist < pt->nearest_dist){



44        pt->nearest = el;



45        pt->nearest_dist = el->dist;



46        pt->nearest_type = el->type;



47    }



48}



49



50static void _ccdPtNearestRenew(ccd_pt_t *pt)



51{



52    ccd_pt_vertex_t *v;



53    ccd_pt_edge_t *e;



54    ccd_pt_face_t *f;



55



56    pt->nearest_dist = CCD_REAL_MAX;



57    pt->nearest_type = 3;



58    pt->nearest = NULL;



59



60    ccdListForEachEntry(&pt->vertices, v, ccd_pt_vertex_t, list){



61        _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)v);



62    }



63



64    ccdListForEachEntry(&pt->edges, e, ccd_pt_edge_t, list){



65        _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)e);



66    }



67



68    ccdListForEachEntry(&pt->faces, f, ccd_pt_face_t, list){



69        _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)f);



70    }



71}

75void ccdPtInit(ccd_pt_t *pt)



76{



77    ccdListInit(&pt->vertices);



78    ccdListInit(&pt->edges);



79    ccdListInit(&pt->faces);



80



81    pt->nearest = NULL;



82    pt->nearest_dist = CCD_REAL_MAX;



83    pt->nearest_type = 3;



84}

86void ccdPtDestroy(ccd_pt_t *pt)



87{



88    ccd_pt_face_t *f, *f2;



89    ccd_pt_edge_t *e, *e2;



90    ccd_pt_vertex_t *v, *v2;



91



92    // first delete all faces



93    ccdListForEachEntrySafe(&pt->faces, f, ccd_pt_face_t, f2, ccd_pt_face_t, list){



94        ccdPtDelFace(pt, f);



95    }



96



97    // delete all edges



98    ccdListForEachEntrySafe(&pt->edges, e, ccd_pt_edge_t, e2, ccd_pt_edge_t, list){



99        ccdPtDelEdge(pt, e);



100    }



101



102    // delete all vertices



103    ccdListForEachEntrySafe(&pt->vertices, v, ccd_pt_vertex_t, v2, ccd_pt_vertex_t, list){



104        ccdPtDelVertex(pt, v);



105    }



106}

109ccd_pt_vertex_t *ccdPtAddVertex(ccd_pt_t *pt, const ccd_support_t *v)



110{



111    ccd_pt_vertex_t *vert;



112



113    vert = CCD_ALLOC(ccd_pt_vertex_t);



114    if (vert == NULL)



115        return NULL;



116



117    vert->type = CCD_PT_VERTEX;



118    ccdSupportCopy(&vert->v, v);



119



120    vert->dist = ccdVec3Len2(&vert->v.v);



121    ccdVec3Copy(&vert->witness, &vert->v.v);



122



123    ccdListInit(&vert->edges);



124



125    // add vertex to list



126    ccdListAppend(&pt->vertices, &vert->list);



127



128    // update position in .nearest array



129    _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)vert);



130



131    return vert;



132}

134ccd_pt_edge_t *ccdPtAddEdge(ccd_pt_t *pt, ccd_pt_vertex_t *v1,



135                                          ccd_pt_vertex_t *v2)



136{



137    const ccd_vec3_t *a, *b;



138    ccd_pt_edge_t *edge;



139



140    if (v1 == NULL || v2 == NULL)



141        return NULL;



142



143    edge = CCD_ALLOC(ccd_pt_edge_t);



144    if (edge == NULL)



145        return NULL;



146



147    edge->type = CCD_PT_EDGE;



148    edge->vertex[0] = v1;



149    edge->vertex[1] = v2;



150    edge->faces[0] = edge->faces[1] = NULL;



151



152    a = &edge->vertex[0]->v.v;



153    b = &edge->vertex[1]->v.v;



154    edge->dist = ccdVec3PointSegmentDist2(ccd_vec3_origin, a, b, &edge->witness);



155



156    ccdListAppend(&edge->vertex[0]->edges, &edge->vertex_list[0]);



157    ccdListAppend(&edge->vertex[1]->edges, &edge->vertex_list[1]);



158



159    ccdListAppend(&pt->edges, &edge->list);



160



161    // update position in .nearest array



162    _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)edge);



163



164    return edge;



165}

167ccd_pt_face_t *ccdPtAddFace(ccd_pt_t *pt, ccd_pt_edge_t *e1,



168                                          ccd_pt_edge_t *e2,



169                                          ccd_pt_edge_t *e3)



170{



171    const ccd_vec3_t *a, *b, *c;



172    ccd_pt_face_t *face;



173    ccd_pt_edge_t *e;



174    size_t i;



175



176    if (e1 == NULL || e2 == NULL || e3 == NULL)



177        return NULL;



178



179    face = CCD_ALLOC(ccd_pt_face_t);



180    if (face == NULL)



181        return NULL;



182



183    face->type = CCD_PT_FACE;



184    face->edge[0] = e1;



185    face->edge[1] = e2;



186    face->edge[2] = e3;



187



188    // obtain triplet of vertices



189    a = &face->edge[0]->vertex[0]->v.v;



190    b = &face->edge[0]->vertex[1]->v.v;



191    e = face->edge[1];



192    if (e->vertex[0] != face->edge[0]->vertex[0]



193            && e->vertex[0] != face->edge[0]->vertex[1]){



194        c = &e->vertex[0]->v.v;



195    }else{



196        c = &e->vertex[1]->v.v;



197    }



198    face->dist = ccdVec3PointTriDist2(ccd_vec3_origin, a, b, c, &face->witness);



199



200



201    for (i = 0; i < 3; i++){



202        if (face->edge[i]->faces[0] == NULL){



203            face->edge[i]->faces[0] = face;



204        }else{



205            face->edge[i]->faces[1] = face;



206        }



207    }



208



209    ccdListAppend(&pt->faces, &face->list);



210



211    // update position in .nearest array



212    _ccdPtNearestUpdate(pt, (ccd_pt_el_t *)face);



213



214    return face;



215}

218void ccdPtRecomputeDistances(ccd_pt_t *pt)



219{



220    ccd_pt_vertex_t *v;



221    ccd_pt_edge_t *e;



222    ccd_pt_face_t *f;



223    const ccd_vec3_t *a, *b, *c;



224    ccd_real_t dist;



225



226    ccdListForEachEntry(&pt->vertices, v, ccd_pt_vertex_t, list){



227        dist = ccdVec3Len2(&v->v.v);



228        v->dist = dist;



229        ccdVec3Copy(&v->witness, &v->v.v);



230    }



231



232    ccdListForEachEntry(&pt->edges, e, ccd_pt_edge_t, list){



233        a = &e->vertex[0]->v.v;



234        b = &e->vertex[1]->v.v;



235        dist = ccdVec3PointSegmentDist2(ccd_vec3_origin, a, b, &e->witness);



236        e->dist = dist;



237    }



238



239    ccdListForEachEntry(&pt->faces, f, ccd_pt_face_t, list){



240        // obtain triplet of vertices



241        a = &f->edge[0]->vertex[0]->v.v;



242        b = &f->edge[0]->vertex[1]->v.v;



243        e = f->edge[1];



244        if (e->vertex[0] != f->edge[0]->vertex[0]



245                && e->vertex[0] != f->edge[0]->vertex[1]){



246            c = &e->vertex[0]->v.v;



247        }else{



248            c = &e->vertex[1]->v.v;



249        }



250



251        dist = ccdVec3PointTriDist2(ccd_vec3_origin, a, b, c, &f->witness);



252        f->dist = dist;



253    }



254}

256ccd_pt_el_t *ccdPtNearest(ccd_pt_t *pt)



257{



258    if (!pt->nearest){



259        _ccdPtNearestRenew(pt);



260    }



261    return pt->nearest;



262}

266{



267    FILE *fout;



268



269    fout = fopen(fn, "a");



270    if (fout == NULL)



271        return;



272



273    ccdPtDumpSVT2(pt, fout);



274



275    fclose(fout);



276}

278void ccdPtDumpSVT2(ccd_pt_t *pt, FILE *fout)



279{



280    ccd_pt_vertex_t *v, *a, *b, *c;



281    ccd_pt_edge_t *e;



282    ccd_pt_face_t *f;



283    int i;



284



285    fprintf(fout, "-----\n");



286



287    fprintf(fout, "Points:\n");



288    i = 0;



289    ccdListForEachEntry(&pt->vertices, v, ccd_pt_vertex_t, list){



290        v->id = i++;



291        fprintf(fout, "%lf %lf %lf\n",



292                ccdVec3X(&v->v.v), ccdVec3Y(&v->v.v), ccdVec3Z(&v->v.v));



293    }



294



295    fprintf(fout, "Edges:\n");



296    ccdListForEachEntry(&pt->edges, e, ccd_pt_edge_t, list){



297        fprintf(fout, "%d %d\n", e->vertex[0]->id, e->vertex[1]->id);



298    }



299



300    fprintf(fout, "Faces:\n");



301    ccdListForEachEntry(&pt->faces, f, ccd_pt_face_t, list){



302        a = f->edge[0]->vertex[0];



303        b = f->edge[0]->vertex[1];



304        c = f->edge[1]->vertex[0];



305        if (c == a || c == b){



306            c = f->edge[1]->vertex[1];



307        }



308        fprintf(fout, "%d %d %d\n", a->id, b->id, c->id);



309    }



310}
