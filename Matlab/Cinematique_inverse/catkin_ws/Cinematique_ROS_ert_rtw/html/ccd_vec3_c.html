82_ccd_inline ccd_real_t __ccdVec3PointSegmentDist2(const ccd_vec3_t *P,



83                                                  const ccd_vec3_t *x0,



84                                                  const ccd_vec3_t *b,



85                                                  ccd_vec3_t *witness)



86{



87    // The computation comes from solving equation of segment:



88    //      S(t) = x0 + t.d



89    //          where - x0 is initial point of segment



90    //                - d is direction of segment from x0 (|d| > 0)



91    //                - t belongs to <0, 1> interval



92    // 



93    // Than, distance from a segment to some point P can be expressed:



94    //      D(t) = |x0 + t.d - P|^2



95    //          which is distance from any point on segment. Minimization



96    //          of this function brings distance from P to segment.



97    // Minimization of D(t) leads to simple quadratic equation that's



98    // solving is straightforward.



99    //



100    // Bonus of this method is witness point for free.



101



102    ccd_real_t dist, t;



103    ccd_vec3_t d, a;



104



105    // direction of segment



106    ccdVec3Sub2(&d, b, x0);



107



108    // precompute vector from P to x0



109    ccdVec3Sub2(&a, x0, P);



110



111    t  = -CCD_REAL(1.) * ccdVec3Dot(&a, &d);



112    t /= ccdVec3Len2(&d);



113



114    if (t < CCD_ZERO || ccdIsZero(t)){



115        dist = ccdVec3Dist2(x0, P);



116        if (witness)



117            ccdVec3Copy(witness, x0);



118    }else if (t > CCD_ONE || ccdEq(t, CCD_ONE)){



119        dist = ccdVec3Dist2(b, P);



120        if (witness)



121            ccdVec3Copy(witness, b);



122    }else{



123        if (witness){



124            ccdVec3Copy(witness, &d);



125            ccdVec3Scale(witness, t);



126            ccdVec3Add(witness, x0);



127            dist = ccdVec3Dist2(witness, P);



128        }else{



129            // recycling variables



130            ccdVec3Scale(&d, t);



131            ccdVec3Add(&d, &a);



132            dist = ccdVec3Len2(&d);



133        }



134    }



135



136    return dist;



137}

139ccd_real_t ccdVec3PointSegmentDist2(const ccd_vec3_t *P,



140                                    const ccd_vec3_t *x0, const ccd_vec3_t *b,



141                                    ccd_vec3_t *witness)



142{



143    return __ccdVec3PointSegmentDist2(P, x0, b, witness);



144}

146ccd_real_t ccdVec3PointTriDist2(const ccd_vec3_t *P,



147                                const ccd_vec3_t *x0, const ccd_vec3_t *B,



148                                const ccd_vec3_t *C,



149                                ccd_vec3_t *witness)



150{



151    // Computation comes from analytic expression for triangle (x0, B, C)



152    //      T(s, t) = x0 + s.d1 + t.d2, where d1 = B - x0 and d2 = C - x0 and



153    // Then equation for distance is:



154    //      D(s, t) = | T(s, t) - P |^2



155    // This leads to minimization of quadratic function of two variables.



156    // The solution from is taken only if s is between 0 and 1, t is



157    // between 0 and 1 and t + s < 1, otherwise distance from segment is



158    // computed.



159



160    ccd_vec3_t d1, d2, a;



161    ccd_real_t u, v, w, p, q, r, d;



162    ccd_real_t s, t, dist, dist2;



163    ccd_vec3_t witness2;



164



165    ccdVec3Sub2(&d1, B, x0);



166    ccdVec3Sub2(&d2, C, x0);



167    ccdVec3Sub2(&a, x0, P);



168



169    u = ccdVec3Dot(&a, &a);



170    v = ccdVec3Dot(&d1, &d1);



171    w = ccdVec3Dot(&d2, &d2);



172    p = ccdVec3Dot(&a, &d1);



173    q = ccdVec3Dot(&a, &d2);



174    r = ccdVec3Dot(&d1, &d2);



175



176    d = w * v - r * r;



177    if (ccdIsZero(d)){



178        // To avoid division by zero for zero (or near zero) area triangles



179        s = t = -1.;



180    }else{



181        s = (q * r - w * p) / d;



182        t = (-s * r - q) / w;



183    }



184



185    if ((ccdIsZero(s) || s > CCD_ZERO)



186            && (ccdEq(s, CCD_ONE) || s < CCD_ONE)



187            && (ccdIsZero(t) || t > CCD_ZERO)



188            && (ccdEq(t, CCD_ONE) || t < CCD_ONE)



189            && (ccdEq(t + s, CCD_ONE) || t + s < CCD_ONE)){



190



191        if (witness){



192            ccdVec3Scale(&d1, s);



193            ccdVec3Scale(&d2, t);



194            ccdVec3Copy(witness, x0);



195            ccdVec3Add(witness, &d1);



196            ccdVec3Add(witness, &d2);



197



198            dist = ccdVec3Dist2(witness, P);



199        }else{



200            dist  = s * s * v;



201            dist += t * t * w;



202            dist += CCD_REAL(2.) * s * t * r;



203            dist += CCD_REAL(2.) * s * p;



204            dist += CCD_REAL(2.) * t * q;



205            dist += u;



206        }



207    }else{



208        dist = __ccdVec3PointSegmentDist2(P, x0, B, witness);



209



210        dist2 = __ccdVec3PointSegmentDist2(P, x0, C, &witness2);



211        if (dist2 < dist){



212            dist = dist2;



213            if (witness)



214                ccdVec3Copy(witness, &witness2);



215        }



216



217        dist2 = __ccdVec3PointSegmentDist2(P, B, C, &witness2);



218        if (dist2 < dist){



219            dist = dist2;



220            if (witness)



221                ccdVec3Copy(witness, &witness2);



222        }



223    }



224



225    return dist;



226}
