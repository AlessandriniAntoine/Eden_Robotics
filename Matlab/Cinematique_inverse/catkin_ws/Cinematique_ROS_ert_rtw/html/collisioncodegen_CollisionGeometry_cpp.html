28    /// compute the support point with interface to libccd



29    void CollisionGeometry::support(const ccd_vec3_t* ccdDir, ccd_vec3_t* ccdSupportVertex) const



30    {



31        // dir should always have non-zero length



32



33        ccd_quat_t quatInv = { {0, 0, 0, 1} };



34        ccdQuatInvert2(&quatInv, &m_quat);



35        ccd_vec3_t dirRot = { {0, 0, 0} };



36        ccd_vec3_t vert = { {0, 0, 0} };



37        



38        ccdVec3Copy(&dirRot, ccdDir);



39        ccdQuatRotVec(&dirRot, &quatInv); // rotate dir into geometry's coordinates



40



41        switch(m_type)



42        {



43        case Type::Box: // for box primitive



44        {



45            ccdVec3Set(&vert, CollisionGeometry::sign(dirRot.v[0])*m_x*CCD_REAL(0.5),



46                CollisionGeometry::sign(dirRot.v[1])*m_y*CCD_REAL(0.5),



47                CollisionGeometry::sign(dirRot.v[2])*m_z*CCD_REAL(0.5));



48            break;



49        }



50        case Type::Sphere:



51        {



52            ccd_real_t len = CCD_SQRT(ccdVec3Len2(&dirRot));



53            ccd_real_t scale = m_radius / len;



54



55            ccdVec3Copy(&vert, &dirRot);



56            ccdVec3Scale(&vert, scale);



57



58            break;



59        }



60        case Type::Cylinder:



61        {



62            ccd_real_t xyShadow = CCD_SQRT(dirRot.v[0] * dirRot.v[0] + dirRot.v[1] * dirRot.v[1]);



63            if (CCD_FABS(xyShadow) < 5 * CCD_EPS) // with double-precision built ccd, CCD_EPS is DBL_EPSILON in float.h, around 2.22045e-16



64            {



65                ccdVec3Set(&vert, CCD_ZERO, CCD_ZERO, sign(dirRot.v[2])*m_height*CCD_REAL(0.5));



66            }



67            else



68            {



69                ccd_real_t scale = m_radius / xyShadow;



70                ccdVec3Set(&vert, scale * dirRot.v[0], scale * dirRot.v[1], sign(dirRot.v[2])*m_height*CCD_REAL(0.5));



71            }



72            break;



73        }



74        case Type::ConvexMesh:



75        {



76            std::vector<ccd_real_t> dotProd;



77            for (auto&& v : m_vertices)



78            {



79                dotProd.push_back(v.v[0] * dirRot.v[0] + v.v[1] * dirRot.v[1] + v.v[2] * dirRot.v[2]);



80            }



81            auto largestItemIter = std::max_element(dotProd.begin(), dotProd.end());



82            std::size_t idx = largestItemIter - dotProd.begin();



83            ccdVec3Set(&vert, m_vertices[idx].v[0], m_vertices[idx].v[1], m_vertices[idx].v[2]);



84            break;



85        }



86        case Type::ConvexMeshFull:



87        default:



88        {



89            // In hill climbing traversal, we can keep track of the indices of visited nodes in a vector



90            // to avoid checking previously visited nodes, but it turns out there is not much performance



91            // gain by doing this, and it is in fact quite costly if we try to recreate this vector



92            // in every support call



93



94            // std::vector<int> visited(m_numVertices, 0)



95



96            ccd_real_t maxValue = m_vertices[0].v[0] * dirRot.v[0] + m_vertices[0].v[1] * dirRot.v[1] + m_vertices[0].v[2] * dirRot.v[2];



97            ccd_real_t val = maxValue;



98            std::size_t currVertIdx = 0;



99            std::size_t nextVertIdx = currVertIdx;



100



101            int numIter = 0;



102            do



103            {



104                currVertIdx = nextVertIdx;



105                for (std::size_t k = 0; k < m_connectionList[currVertIdx].size(); k++)



106                {



107                    std::size_t idx = static_cast<std::size_t>(m_connectionList[currVertIdx][k]);



108                    val = ccdVec3Dot( &m_vertices[idx], &dirRot );



109                    if (val > maxValue)



110                    {



111                        maxValue = val;



112                        nextVertIdx = m_connectionList[currVertIdx][k];



113                    }



114                    numIter++;



115



116                }



117



118            } while (currVertIdx != nextVertIdx);



119            //std::cout << "numIter = " << numIter << std::endl



120            ccdVec3Set(&vert, m_vertices[currVertIdx].v[0], m_vertices[currVertIdx].v[1], m_vertices[currVertIdx].v[2]);



121            



122        }



123



124        }



125



126        // transform support point to world coordinates (quat * vert + pos)



127        ccdQuatRotVec(&vert, &m_quat);



128        ccdVec3Add(&vert, &m_pos);



129        ccdVec3Copy(ccdSupportVertex, &vert);



130



131    }

133    /// 



134    void CollisionGeometry::constructConnectionList()



135    {



136        std::cout << "construct connection list" << std::endl;



137        // initialize



138        m_connectionList.resize(m_numVertices);



139        for (std::size_t i = 0; i < m_numVertices; i++)



140        {



141            m_connectionList[i].resize(16);



142            m_connectionList[i].clear();



143        }



144



145        // populate



146        for (std::size_t j = 0; j < m_numFaces; j++)



147        {



148            m_connectionList[m_faces[j].v[0]].push_back(m_faces[j].v[1]);



149            m_connectionList[m_faces[j].v[0]].push_back(m_faces[j].v[2]);



150            m_connectionList[m_faces[j].v[1]].push_back(m_faces[j].v[0]);



151            m_connectionList[m_faces[j].v[1]].push_back(m_faces[j].v[2]);



152            m_connectionList[m_faces[j].v[2]].push_back(m_faces[j].v[0]);



153            m_connectionList[m_faces[j].v[2]].push_back(m_faces[j].v[1]);



154        }



155



156        // reduce



157        for (std::size_t i = 0; i < m_numVertices; i++)



158        {



159            std::sort(m_connectionList[i].begin(), m_connectionList[i].end());



160            auto last = std::unique(m_connectionList[i].begin(), m_connectionList[i].end());



161            m_connectionList[i].erase(last, m_connectionList[i].end());



162        }



163    }
