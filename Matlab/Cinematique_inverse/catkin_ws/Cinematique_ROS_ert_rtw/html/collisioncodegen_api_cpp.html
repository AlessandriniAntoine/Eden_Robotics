32int collisioncodegen_intersect(CollisionGeometryType obj1,



33                              double* pos1, 



34                              double* quat1,



35                              CollisionGeometryType obj2,



36                              double* pos2, 



37                              double* quat2,



38                              double computeDistance,



39                              double* p1Vec,



40                              double* p2Vec,



41                              double* distance)



42{



43    shared_robotics::updatePose(obj1, pos1, quat1);



44    shared_robotics::updatePose(obj2, pos2, quat2);



45    return shared_robotics::intersect(obj1, obj2, static_cast<int>(computeDistance), p1Vec, p2Vec, *distance);



46}
