18ccd_real_t shared_robotics::distanceToOrigin(ccd_simplex_t * simplex, ccd_vec3_t *& closestPoint)



19{



20    // find the distance from origin to the given simplex



21



22    ccd_real_t distSq;



23    ccd_real_t distance = 0;



24



25    switch (ccdSimplexSize(simplex)) 



26    {



27    case 1: // point



28    {



29        closestPoint->v[0] = simplex->ps[0].v.v[0];



30        closestPoint->v[1] = simplex->ps[0].v.v[1];



31        closestPoint->v[2] = simplex->ps[0].v.v[2];



32        distSq = ccdVec3Len2(closestPoint);



33        distance = CCD_SQRT(distSq);



34        break;



35    }



36    case 2: // line segment



37    {



38        ccd_vec3_t p0 = simplex->ps[0].v;



39        ccd_vec3_t p1 = simplex->ps[1].v;



40        distSq = ccdVec3PointSegmentDist2(ccd_vec3_origin, &p0, &p1, closestPoint);



41        distance = CCD_SQRT(distSq);



42        break;



43    }



44    case 3: // triangle



45    {



46        ccd_vec3_t p0 = simplex->ps[0].v;



47        ccd_vec3_t p1 = simplex->ps[1].v;



48        ccd_vec3_t p2 = simplex->ps[2].v;



49        distSq = ccdVec3PointTriDist2(ccd_vec3_origin, &p0, &p1, &p2, closestPoint);



50        distance = CCD_SQRT(distSq);



51        break;



52    }



53    default: // simplex can be a tetrahedron. 



54             // The simplex generated by GJK is not necessarily sufficiently refined/reduced to report the actual distance



55    {



56        // cycle through all faces in the tetrahedron and find the one with shortest distance to origin



57



58        const std::size_t indices[4][3] = { { 3,1,2 },{ 0,3,2 },{ 0,1,3 },{ 0,1,2 }}; // >> combnk([0 1 2 3], 3)



59        ccd_vec3_t closestPointCandidate;



60



61        ccd_real_t bestDistSq = -1;



62        std::size_t bestIdx = 0;



63        for (std::size_t i = 0; i <= 3; i++)



64        {



65            ccd_vec3_t p0 = simplex->ps[indices[i][0]].v;



66            ccd_vec3_t p1 = simplex->ps[indices[i][1]].v;



67            ccd_vec3_t p2 = simplex->ps[indices[i][2]].v;



68



69            distSq = ccdVec3PointTriDist2(ccd_vec3_origin, &p0, &p1, &p2, &closestPointCandidate);



70            if (i == 0)



71            {



72                bestDistSq = distSq;



73                bestIdx = 0;



74                closestPoint->v[0] = closestPointCandidate.v[0];



75                closestPoint->v[1] = closestPointCandidate.v[1];



76                closestPoint->v[2] = closestPointCandidate.v[2];



77            }



78            else if (distSq < bestDistSq)



79            {



80                bestDistSq = distSq;



81                bestIdx = i;



82                closestPoint->v[0] = closestPointCandidate.v[0];



83                closestPoint->v[1] = closestPointCandidate.v[1];



84                closestPoint->v[2] = closestPointCandidate.v[2];



85            }



86        }



87        distance = CCD_SQRT(bestDistSq);



88        // also reduce tetrahedron to triangle



89        if (bestIdx < 3)



90        {



91            simplex->ps[bestIdx] = simplex->ps[3];



92        }



93        simplex->last = 2;



94    }



95    }



96



97    return distance;



98}

101void shared_robotics::extractWitnessPoints(const ccd_simplex_t *simplex, const ccd_vec3_t *closestPoint, ccd_vec3_t *p1, ccd_vec3_t *p2)



102{



103    // witness points: a pair of points from the two objects that realize the minimum separation distance between them.



104    switch (ccdSimplexSize(simplex))



105    {



106    case 1:// point



107    {



108        *p1 = simplex->ps[0].v1;



109        *p2 = simplex->ps[0].v2;



110        break;



111    }



112    case 2: // line segment



113    {



114        interpolateBetweenTwoSupportPoints(simplex, closestPoint, p1, p2);



115        break;



116    }



117    case 3: // triangle



118    {



119        interpolateAmongThreeSupportPoints(simplex, closestPoint, p1, p2);



120        break;



121    }



122    }



123



124}

126void shared_robotics::interpolateBetweenTwoSupportPoints(const ccd_simplex_t *& simplex, const ccd_vec3_t *& closestPoint, ccd_vec3_t *& p1, ccd_vec3_t *& p2)



127{



128    // interpolate between two support points m, n



129    //



130    // o------------o-------------o



131    // m(m1,m2)     p(p1,p2)      n(n1,n2)



132    //



133    ccd_vec3_t m, n, p, dp, dn;



134    ccdVec3Copy(&m, &simplex->ps[0].v);



135    ccdVec3Copy(&n, &simplex->ps[1].v);



136    ccdVec3Copy(&p, closestPoint);



137



138    ccdVec3Sub2(&dp, &p, &m);



139    ccdVec3Sub2(&dn, &n, &m);



140    auto dpm = CCD_SQRT(ccdVec3Len2(&dp));



141    auto dnm = CCD_SQRT(ccdVec3Len2(&dn));



142



143    if (dnm < std::numeric_limits<ccd_real_t>::epsilon())



144    {



145        *p1 = simplex->ps[0].v1;



146        *p2 = simplex->ps[0].v2;



147    }



148    else



149    {



150        ccd_vec3_t m1, m2, n1, n2, tmp, tmp2;



151        ccdVec3Copy(&m1, &simplex->ps[0].v1);



152        ccdVec3Copy(&m2, &simplex->ps[0].v2);



153        ccdVec3Copy(&n1, &simplex->ps[1].v1);



154        ccdVec3Copy(&n2, &simplex->ps[1].v2);



155



156        auto s = dpm / dnm;



157        ccdVec3Sub2(&tmp, &n1, &m1);



158        ccdVec3Scale(&tmp, s);



159        ccdVec3Add(&tmp, &m1);



160        



161        ccdVec3Copy(p1, &tmp);



162



163        ccdVec3Sub2(&tmp2, &n2, &m2);



164        ccdVec3Scale(&tmp2, s);



165        ccdVec3Add(&tmp2, &m2);



166        



167        ccdVec3Copy(p2, &tmp2);



168    }



169}

171/// convert a point's position in global coordinates to the barycentric coordinates defined by three points in global coordinates (internal use only).



172void barycentricCoordinates(const ccd_vec3_t * a, const ccd_vec3_t * b, const ccd_vec3_t * c, const ccd_vec3_t * p, ccd_vec3_t * lambda)



173{



174    // a, b, c must not be co-linear



175    ccd_vec3_t ab, ac, bc, bp, cp, n, na, nb, neg_ac;



176    ccdVec3Sub2(&ab, b, a);



177    ccdVec3Sub2(&ac, c, a);



178    ccdVec3Sub2(&bc, c, b);



179



180    ccdVec3Sub2(&bp, p, b);



181    ccdVec3Sub2(&cp, p, c);



182



183    ccdVec3Cross(&n, &ab, &ac);



184    ccdVec3Cross(&na, &bc, &bp);



185    ccdVec3Copy(&neg_ac, &ac);



186    ccdVec3Scale(&neg_ac, -1);



187    ccdVec3Cross(&nb, &neg_ac, &cp);



188



189



190    ccd_real_t nn = ccdVec3Dot(&n, &n);



191    ccd_real_t alpha = ccdVec3Dot(&n, &na) / nn;



192    ccd_real_t beta = ccdVec3Dot(&n, &nb) / nn;



193



194    ccdVec3Set(lambda, alpha, beta, 1 - alpha - beta);



195}

197void shared_robotics::interpolateAmongThreeSupportPoints(const ccd_simplex_t *& simplex, const ccd_vec3_t *& closestPoint, ccd_vec3_t *& p1, ccd_vec3_t *& p2)



198{



199    /* interpolate between three support points a, b, c



200



201               o a(a1,a2)



202              / \



203             /   \



204            /     \



205           /       \



206          /  o      \



207         /   p(p1,p2)\



208        o-------------o



209        b(b1,b2)       c(c1,c2)



210    */



211



212    ccd_vec3_t a, b, c, p, baVec, caVec, cbVec, crprod;



213    ccdVec3Copy(&a, &simplex->ps[0].v);



214    ccdVec3Copy(&b, &simplex->ps[1].v);



215    ccdVec3Copy(&c, &simplex->ps[2].v);



216    ccdVec3Copy(&p, closestPoint);



217    



218    ccdVec3Sub2(&baVec, &b, &a);



219    ccdVec3Sub2(&caVec, &c, &a);



220    ccdVec3Cross(&crprod, &baVec, &caVec);



221



222    ccdVec3Sub2(&cbVec, &c, &b);



223



224    if ( CCD_SQRT(ccdVec3Len2(&crprod)) < std::numeric_limits<ccd_real_t>::epsilon()) // if triangle area is zero



225    {



226        ccd_simplex_t simplexTmp;



227        ccd_real_t ab = CCD_SQRT(ccdVec3Len2(&baVec));



228        ccd_real_t ac = CCD_SQRT(ccdVec3Len2(&caVec));



229        ccd_real_t bc = CCD_SQRT(ccdVec3Len2(&cbVec));



230



231        simplexTmp.last = 1;



232        if (ab >= ac && ab >= bc)



233        {



234            simplexTmp.ps[0] = simplex->ps[0];



235            simplexTmp.ps[1] = simplex->ps[1];



236        }



237        else if (ac > ab && ac > bc)



238        {



239            simplexTmp.ps[0] = simplex->ps[0];



240            simplexTmp.ps[1] = simplex->ps[2];



241        }



242        else



243        {



244            simplexTmp.ps[0] = simplex->ps[1];



245            simplexTmp.ps[1] = simplex->ps[2];



246        }



247



248        const ccd_simplex_t * simplexPtr = &simplexTmp;



249        interpolateBetweenTwoSupportPoints(simplexPtr, closestPoint, p1, p2);



250    }



251    else



252    {



253        ccd_vec3_t lambda = { {0, 0, 0} };



254        barycentricCoordinates(&a, &b, &c, &p, &lambda);



255



256        ccd_vec3_t a1, a2, b1, b2, c1, c2;



257        ccd_vec3_t tmp1 = { {0, 0, 0} };



258        ccd_vec3_t tmp2 = { {0, 0, 0} };



259        ccdVec3Copy(&a1, &simplex->ps[0].v1);



260        ccdVec3Copy(&a2, &simplex->ps[0].v2);



261



262        ccdVec3Copy(&b1, &simplex->ps[1].v1);



263        ccdVec3Copy(&b2, &simplex->ps[1].v2);



264



265        ccdVec3Copy(&c1, &simplex->ps[2].v1);



266        ccdVec3Copy(&c2, &simplex->ps[2].v2);



267



268        ccdVec3Scale(&a1, lambda.v[0]);



269        ccdVec3Scale(&b1, lambda.v[1]);



270        ccdVec3Scale(&c1, lambda.v[2]);



271



272        ccdVec3Add(&tmp1, &a1);



273        ccdVec3Add(&tmp1, &b1);



274        ccdVec3Add(&tmp1, &c1);



275



276        ccdVec3Copy(p1, &tmp1);



277



278        ccdVec3Scale(&a2, lambda.v[0]);



279        ccdVec3Scale(&b2, lambda.v[1]);



280        ccdVec3Scale(&c2, lambda.v[2]);



281



282        ccdVec3Add(&tmp2, &a2);



283        ccdVec3Add(&tmp2, &b2);



284        ccdVec3Add(&tmp2, &c2);



285



286        ccdVec3Copy(p2, &tmp2);



287    }



288



289}

293ccd_real_t shared_robotics::ccdDistance(const void *obj1, const void *obj2, const ccd_t *ccd,



294                ccd_vec3_t* p1, ccd_vec3_t* p2)



295{



296    ccd_simplex_t simplex; // double check



297    ccd_vec3_t closestPoint; // placeholder



298    ccd_real_t distance = -1;



299    ccd_real_t lastDistance;



300



301    ccd_simplex_t * simplexPtr = &simplex;



302    ccd_vec3_t *closestPointPtr = &closestPoint; 



303



304    if (__ccdGJK(obj1, obj2, ccd, &simplex) == 0) // __ccdGJK return -1 if intersection is not found or 0 if intersection is found



305    {



306        return -CCD_ONE;



307    }



308    



309    int maxIterations = static_cast<int>(ccd->max_iterations);



310    lastDistance = CCD_REAL_MAX;



311



312    // if the input simplex comes out from __ccdGJK,



313    // in many cases, the for-loop should exit after the first iteration



314    for (int i = 0; i < maxIterations; i++)  



315    {



316



317        distance = distanceToOrigin(simplexPtr, closestPointPtr);



318



319        if (CCD_FABS(lastDistance - distance) < ccd->dist_tolerance) // if no further improvement



320        {



321            // interpolate to populate p1, p2



322            extractWitnessPoints(simplexPtr, closestPointPtr, p1, p2);



323            return distance;



324        }



325        



326        lastDistance = distance;



327



328        // further examination



329        // validate the closest point through support direction



330        // compute the support direction based on the direction as determined by the current closestPoint 



331        ccd_vec3_t dir = { -closestPointPtr->v[0], -closestPointPtr->v[1], -closestPointPtr->v[2] };



332        ccd_support_t closestPointNew;



333        __ccdSupport(obj1, obj2, &dir, ccd, &closestPointNew);



334        distance = CCD_SQRT(ccdVec3Len2(&closestPointNew.v));



335



336        // if the acquired support point deviates from the closestPoint candidate



337        // add the support point to the simplex and rerun distance function



338        // but we need to first check if the new support point coincides with any existing ones



339        



340        int samePoint = 0;



341        for (std::size_t q = 0; q <= static_cast<std::size_t>(simplex.last); q++)



342        {



343            samePoint = samePoint || ccdVec3Eq(&closestPointNew.v, &simplex.ps[q].v);



344        }



345        



346        if (!samePoint)



347        {



348            simplex.ps[simplex.last + 1] = closestPointNew;



349            simplex.last += 1;



350        }



351



352        // for debugging, uncomment the following lines to step through the simplex migration 



353        //std::cout << "----------Iter " << i << "-------------" << std::endl;



354        //for (std::size_t q = 0; q <= simplex.last; q++)



355        //{



356        //    std::cout << "simplex ps[" << q << "] = [ " << simplex.ps[q].v.v[0] << " " << simplex.ps[q].v.v[1] << " " << simplex.ps[q].v.v[2] << " ] " << std::endl;



357        //}



358



359    }



360



361    // reach iteration limit



362    extractWitnessPoints(simplexPtr, closestPointPtr, p1, p2);



363    return distance;



364    



365}
