19EXTERN_C COLLISIONCODEGEN_API int shared_robotics::intersect(const void* obj1, const void* obj2, int computeDistance, ccd_real_t* p1Vec, ccd_real_t* p2Vec, ccd_real_t& distance)



20{



21        ccd_t ccd;



22        CCD_INIT(&ccd); // initialize ccd_t struct



23        ccd_support_fn f = &fSupport;



24        ccd.support1 = f;



25        ccd.support2 = f;



26        ccd.max_iterations = 100; // maximal number of iterations



27



28        int result;



29        ccd_vec3_t p1 = {{0, 0, 0}};



30        ccd_vec3_t p2 = {{0, 0, 0}};



31        distance = -CCD_ONE;



32        p1Vec[0] = 0;



33        p1Vec[1] = 0;



34        p1Vec[2] = 0;



35



36        p2Vec[0] = 0;



37        p2Vec[1] = 0;



38        p2Vec[2] = 0;



39



40        // std::cout << "computeDistance = " <<computeDistance << std::endl;



41        if (computeDistance) 



42        {



43            // returns -1 if two bodies intersect, otherwise returns the minimal distance



44            distance = ccdDistance(obj1, obj2, &ccd, &p1, &p2);



45



46            p1Vec[0] = p1.v[0];



47            p1Vec[1] = p1.v[1];



48            p1Vec[2] = p1.v[2];



49



50            p2Vec[0] = p2.v[0];



51            p2Vec[1] = p2.v[1];



52            p2Vec[2] = p2.v[2];



53



54            if (distance < 0) 



55            {



56                result = 1;



57            } 



58            else 



59            {



60                result = 0;



61            }



62        } 



63        else 



64        {



65            // returns 1 if two objects intersect



66            result = ccdGJKIntersect(obj1, obj2, &ccd);



67        }



68



69        return result;



70}

72EXTERN_C COLLISIONCODEGEN_API void shared_robotics::updatePose(void* obj, const ccd_real_t* position, const ccd_real_t* orientation) 



73{



74    auto geom = static_cast<CollisionGeometry*>(obj);



75    ccdVec3Set(&geom->m_pos, static_cast<ccd_real_t>(position[0]), static_cast<ccd_real_t>(position[1]), static_cast<ccd_real_t>(position[2]));



76    ccdQuatSet(&geom->m_quat, static_cast<ccd_real_t>(orientation[1]), static_cast<ccd_real_t>(orientation[2]), static_cast<ccd_real_t>(orientation[3]), static_cast<ccd_real_t>(orientation[0])); // x,y,z,w in ccd



77}

79static void fSupport(const void* obj, const ccd_vec3_t* dir, ccd_vec3_t* supportPoint)



80{



81    auto geomPtr = (static_cast<const CollisionGeometry*>(obj));



82    geomPtr->support(dir, supportPoint);



83}
